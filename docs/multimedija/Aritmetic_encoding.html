Aritmetično kodiranje

Aritmeticni kodirnik se v multimediji pogosto uporablja pri brezizgubnem stiskanju podatkovnih slik, zvoka in teksta. Gre za matematično dokazano najbolj unčikovito kodiranje če se omejimo le na simbole velikosti 8 bitov. Veliko bolj so unčikovito stiskanje dosezemo z zip, ki pa hrani bistveno daljše simbole.

Kodiranje datoteke opravljamo v naslednjih korakih.

Korak 1. štetje simbolov

    //create table storing all symbols
    let mut table_frequency:[u64;ENCODER_SYMBOL_TABLE_SIZE] = [0;ENCODER_SYMBOL_TABLE_SIZE];
    let mut table_lower_limit:[u64;ENCODER_SYMBOL_TABLE_SIZE] = [0;ENCODER_SYMBOL_TABLE_SIZE];
    let mut table_upper_limit:[u64;ENCODER_SYMBOL_TABLE_SIZE] = [0;ENCODER_SYMBOL_TABLE_SIZE];

    //println!("{:?}bit encoder {:?} posible symbols",ENCODER_BITS,ENCODER_SYMBOL_TABLE_SIZE);
    let mut buff:[u8;8] = [0;8]; //usize has 8 bytes

    //println!("Counting ocuurances...");
    //f_in.filestram.seek(SeekFrom::Start(0)).expect("cant find the start of the file");
    let mut ocuured_symbols: Vec<usize> = Vec::new();
    let mut max_ocurrances: u64 = 0;
    let mut frequency_bits: usize = 0;
    {
        let mut f_in:BinFile = BinFile::new_read(filename_in);
        loop {
            let res:usize = f_in.read_n_be_bits_to_8_byte_buff(&ENCODER_SYMBOL_BITS, &mut buff, true);
            if res < 1 {break;}
            let symbol:usize = usize::from_be_bytes(buff);

            //DEBUG
            //println!("symbol:{:?}",symbol);

            if table_frequency[symbol] == 0 { ocuured_symbols.push(symbol); }
            table_frequency[symbol] += 1;
            if table_frequency[symbol] > max_ocurrances {max_ocurrances = table_frequency[symbol];}
        }
        max_ocurrances = max_ocurrances-1;
        let mut max: u64 = max_ocurrances;
        while max != 0 {
            max = max/2;
            frequency_bits += 1;
        }
        //println!("max num of ocurrances: {:?} can be stored in {:?} bits",max_ocurrances,frequency_bits);
    }

korak 2. hranimo pojavitve v glavo stisjene datoteke

    //write file header
    //frequency_bits = the smallest posible number of bits representing the max frequency
    //freqency 0 is frequency 1 becuse we dont write symbols in the header with frequency of 0

    f_encoded.write_n_bits_from_be_buff(&8, & frequency_bits.to_be_bytes());
    f_encoded.write_n_bits_from_be_buff(&ENCODER_SYMBOL_BITS, & (ocuured_symbols.len()-1).to_be_bytes());

Glava datoteke je sestvljena naslednje. Najprej zapisemo 8 bitov ki predstavljajo vrednost F. F nam pove z koliko
bitov zapisujemo stevila pojavitev v glavi. Nato z F biti zapisemo stevilo simbolov, ki jih enkodiramo.
Na koncu sledi zapis vseh 8bitnih simbolov z F bitnim stevilom pojavitev.

F = [8](stevilo bitov potrebnih za zapis pojavitev simbola)
N = [F](koliko simbolov imamo zapisanih v glavi)
simbol_1 = [8](simbol) + [F](stevilo pojavitev simbola zapisan z F biti)
...
simbol_n = [8](simbol) + [F](stevilo pojavitev simbola zapisan z F biti)

korak 3. pridobi zgornjo in spodnjo mejo

    //calculate lower and upper limit
    let mut lower_limit:u64 = 0;
    let mut upper_limit:u64 = 0;

    for i in &ocuured_symbols{
        f_encoded.write_n_bits_from_be_buff(&ENCODER_SYMBOL_BITS, &i.to_be_bytes()); //symbol
        f_encoded.write_n_bits_from_be_buff(&frequency_bits, &(table_frequency[*i]-1).to_be_bytes()); //frequency of that symbol

        upper_limit = lower_limit+table_frequency[*i];

        table_lower_limit[*i] = lower_limit;
        table_upper_limit[*i] = upper_limit;

        lower_limit = upper_limit;
    }

korak 4. iniciliziraj enkodiranje

    //inicilize encoding
    let cumulative_frequency:u64 = upper_limit.clone(); 
    lower_limit = 0;
    upper_limit = 2_u64.pow(ENCODER_BITS as u32 -1)-1;
    let second_fourth:u64 = (upper_limit+1)/2;
    let first_fourth:u64 = second_fourth/2;
    let third_fourth:u64 = first_fourth *3;
    let mut e3_counter:u64 = 0;

korak 5. kodiraj znake z huffmanovimi kodami

    let mut f_in:BinFile = BinFile::new_read(filename_in);
    loop {
        let res:usize = f_in.read_n_be_bits_to_8_byte_buff(&ENCODER_SYMBOL_BITS, &mut buff, true);
        if res < 1 {break;}
        let symbol:usize = usize::from_be_bytes(buff);
        let symbol_low:u64 = table_lower_limit[symbol];
        let symbol_high:u64 = table_upper_limit[symbol];
        let step:u64 = (upper_limit-lower_limit+1)/cumulative_frequency;
        let step_L:u64 = ((upper_limit-lower_limit+1)*symbol_low)/cumulative_frequency;
        let step_H:u64 = ((upper_limit-lower_limit+1)*symbol_high)/cumulative_frequency;
        //print!("symbol:{:?} H:{:?} L{:?} step:{:?} ",symbol,symbol_high,symbol_low,step);

        upper_limit = lower_limit + (step * symbol_high) - 1;
        lower_limit = lower_limit + step * symbol_low;

        //DEBUG
        //print!("lim(H:{:?} L:{:?}) ",upper_limit,lower_limit);

        while upper_limit < second_fourth || lower_limit >= second_fourth {
            if upper_limit < second_fourth {
                lower_limit = lower_limit*2;
                upper_limit = (upper_limit*2)+1;
                f_encoded.write_bit(0); //encoded_bits += "0";
                //print!("E1(L:{:?}H:{:?}) ",lower_limit,upper_limit);
                for _ in 0..e3_counter {f_encoded.write_bit(1); /*encoded_bits += "1";*/}
                e3_counter = 0;
            }
            if lower_limit >= second_fourth {
                lower_limit = 2 * (lower_limit - second_fourth);
                upper_limit = 2 * (upper_limit - second_fourth) + 1;
                f_encoded.write_bit(1);//encoded_bits += "1";
                //print!("E2(L:{:?}H:{:?}) ",lower_limit,upper_limit);
                for _ in 0..e3_counter {f_encoded.write_bit(0); /*encoded_bits += "0";*/}
                e3_counter = 0;
            }
        }
        while lower_limit >= first_fourth && upper_limit < third_fourth {
            lower_limit = 2 * (lower_limit - first_fourth);
            upper_limit = 2 * (upper_limit - first_fourth) + 1;
            //print!("E3(L:{:?}H:{:?}) ",lower_limit,upper_limit);
            e3_counter += 1;
            
        }
    }
    if lower_limit < first_fourth {
        f_encoded.write_bit(0); //encoded_bits += "0";
        f_encoded.write_bit(1); //encoded_bits += "1";
        for _ in 0..e3_counter {f_encoded.write_bit(1); /*encoded_bits += "1";*/}
    }
    else {
        f_encoded.write_bit(1); //encoded_bits += "1";
        f_encoded.write_bit(0); //encoded_bits += "0";
        for _ in 0..e3_counter {f_encoded.write_bit(0); /*encoded_bits += "1";*/}
    }
    f_encoded.file_close();
